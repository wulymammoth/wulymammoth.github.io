JavaScript

class Address {
  constructor(street, state, country) {
    this.street = street;
    this.state = state;
    this.country = "United States of America";
  }
}

class Person {
  constructor(name, address) {
    this.name = name;
    this.address = address;
  }
}

const renderingMixin = {
  render() {
    console.log(this);
  }
};

// include mixin in each class
[Address, Person].forEach(klass => {
  Object.assign(klass.prototype, renderingMixin);
});
